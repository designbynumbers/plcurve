\begin{frame}
  \frametitle{How to enumerate knot diagrams (like a topologist)}
  \begin{definition}
    A \emph{knot shadow} is a equivalence class of generic immersions of the unoriented $S^1$
    into the sphere $S^2$ up to diffeomorphism of $S^2$.
  \end{definition}

  \begin{plan}
   \begin{enumerate}
   \item Enumerate shadows (and discard isomorphic shadows)
   \item Assign crossing and orientation information (and discard crossing patterns related by an automorphism of the shadow)
   \end{enumerate}
  \end{plan}

  \begin{observationfirst}<2->
  Symmetry stinks.
  \end{observationfirst}
\end{frame}

\begin{frame}
  \frametitle{Tabulating knot shadows: \texttt{plantri}, two ways}
  \begin{proposition}
    Knot shadows $\leftrightarrow$ 1-component 4-valent embedded

\begin{frame} 
   \frametitle{SQS: How to get everyone working}
   
   SQS is a message queue-- the messages describe work to do. You can write to and read from an sqs queue with the Python library boto. 
   
   \begin{itemize}
   \item Worker reads message (should describe $< 1$ hour of work). 
   \item SQS ``locks'' the message for one hour. 
   \item If the worker completes the job, the worker should delete the message.
   \item If the worker dies, the message unlocks and is returned to the queue. 
   \item Messages returned to queue more than $n$ times are set aside for debugging.
   \end{itemize}
   
   Your first million messages (per month) are free. SQS is distributed, backed up, fault-tolerant and always working.
\end{frame}

\begin{frame}
  \frametitle{A question on unknotting}
  \begin{theorem}[[Frisch-Wassermann-Delbr\"uck Conjecture{]} Sumners-Whittington 1988]
    The ratio of unknots in random $n$-edge self-avoiding lattice
    polygons tends to zero exponentially with $n$.
  \end{theorem}
  \vspace{.25in}
  \begin{theorem}[Chapman]
    The ratio of unknots in diagrams tends to zero exponentially as $n$
    increases. 
  \end{theorem}
  
\end{frame}
